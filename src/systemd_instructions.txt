Create a new systemd unit file at /etc/system/systemd/kettlewright.service:

[Unit]
Description=Start Kettlewright Gunicorn Server
After=network.target

[Service]
User=kettlewright
Group=kettlewright
WorkingDirectory=/home/kettlewright/app/kettlewright/
EnvironmentFile=/home/kettlewright/app/.env
Environment="PATH=/home/kettlewright/.local/share/virtualenvs/kettlewright-zHfuFjWL/bin"
ExecStart=/home/kettlewright/.local/share/virtualenvs/kettlewright-zHfuFjWL/bin/gunicorn -k eventlet -w 2 -b 0.0.0.0:8000 'app:application'

[Install]
WantedBy=multi-user.target

---

2. Reload systemd units:
systemctl daemon-reload

3. Start the service:
systemctl start kettlewright

3. Enable it on boot:
systemctl enable kettlewright
